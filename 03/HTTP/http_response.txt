<script>
$(document).ready(function() {
    var vals = location.search.replace('?', '').split('=');
    if(vals.length > 1) {
      var vals = vals[1];
      vals = decodeURIComponent(atob(vals))
      var questions = vals.split(',');

      for(var question of questions) { // for each question
        if($('#' + question).length) { // exists
          if($('#' + question).is("ul")) { // highlight entire chapter
            $('#' + question).children().addClass('highlight');
            question = question.replace('c', 'h');
            $('#' + question).addClass('highlight');
          }
          else {
            $('#' + question).addClass('highlight');
          }
        }
      }

      if(questions.length > 0) {
        var q = questions[0];
        if($('#' + q).length) {
          document.getElementById(q).scrollIntoView({block: 'center'});
        }
      }
    }
    return;
});

$(document).on('click', '#create-highlights', function() {
  /*
    When clicked, checkbox inputs will be added to each problem and
    chapter heading, and the 'generate' button will be unhidden
  */
  $('#gen-highlights').removeClass('d-none');
  $(this).addClass('d-none');

  var children = $('#problems').children();
  for(var i = 0; i < children.length; i++) {
    var child = children[i];
    if(child.tagName == 'H5') {
      $('<input class=\'form-check-input\' type=\'checkbox\' value=\'\' id=\'checkbox-' + $(child).attr('id') + '\'>').insertBefore(child);
    }
    else if(child.tagName == 'UL') {
      var ul_children = $(child).children();
      for(var j = 0; j < ul_children.length; j++) {
        var ul_child = ul_children[j];
        $(ul_child).prepend('<input class=\'form-check-input\' type=\'checkbox\' value=\'\' id=\'checkbox-' + $(ul_child).attr('id') + '\'>');
      }
    }
  }
});

$(document).on('click', "input", function() {
    /*
      When one of the Chapter checkboxes is clicked, it will select all problems and
      disable changes to checkboxes in said chapter
    */
    if($(this).attr('id').startsWith('checkbox-h')) {
      var subproblems = $('#c'+$(this).attr('id').slice(-1)).children();
      for(var i = 0; i < subproblems.length; i++) {
        var subproblem = $(subproblems[i]).find("input");
        if($(this).is(':checked')) {
          $(subproblem).prop('indeterminate', true);
          $(subproblem).prop('checked', false);
          $(subproblem).prop('disabled', true);
        }
        else {
          $(subproblem).prop('indeterminate', false);
          $(subproblem).prop('checked', false);
          $(subproblem).prop('disabled', false);
        }
      }
    }
});

$(document).on('click', '#gen-highlights', function() {
  /*
    When clicked, highlight string will be generated, this button will be
    hidden, and checkbox inputs will be removed from the page
  */

  var output = "";

  $("input:checked").each(function() {
    var stripped = $(this).attr('id').replace('checkbox-', '');
    if($('#'+stripped).is("h5")) {
      stripped = stripped.replace('h', 'c');
    }
    output += stripped + ",";
  });

  output = output.slice(0, -1);
  output = encodeURIComponent(btoa(output).replaceAll('=', ''));

  // jump to page
  window.location.href = "index.php?p="+output;

});
</script>
